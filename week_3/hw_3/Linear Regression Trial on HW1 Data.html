class LinearRegression():
    
    def __init__(self, input_dim, output_dim, learning_rate = 0.00001, epochs = 20000, display_per_epochs = 100):
#         self.w = np.random.rand((input_dim + 1))
        self.w = np.zeros((input_dim + 1))
#         print(self.w)
        self.learning_rate = learning_rate
        self.epochs = epochs
        self.display_per_epochs = display_per_epochs
        self.history = {
            'loss_train': [],
            'acc_train': []
        }
        
    def train(self, x, y):
        x = prepand_one(x)
#         print(self.w.shape)
        self.history['loss_train'].append(self.loss(x, y))
        print('Epoch %i loss: '%0, self.history['loss_train'][-1])
        for i in range(1, self.epochs + 1):
            random_indices = np.random.permutation(x.shape[0])
            for x_in, y_out in zip(x[random_indices], y[random_indices]):
#             for x_in, y_out in zip(x, y):
#                 pre_loss = self.loss(x_in, y_out)
                self.w = self.w - self.learning_rate * self.step_gradient(x_in, y_out)
#                 after_loss = self.loss(x_in, y_out)
#                 print(pre_loss - after_loss)
#                 print('-'*50)
            if i%self.display_per_epochs == 0:
                self.history['loss_train'].append(self.loss(x, y))
                print('Epoch %i loss: '%i, self.loss(x, y))
            
        
    def step_gradient(self, x, y):
        return -(2/x.shape[0])*(y - np.matmul(self.w, x.T))*x
    
    def inference(self, x):
        return np.matmul(self.w, x.T).reshape((x.shape[0], y_dim))
    
    def loss(self, x, y):
#         print((1/x.shape[0])*np.sum(np.power(y - self.predict(x), 2)))
        return (1/x.shape[0])*np.sum(np.power(y - self.inference(x), 2))
    
    def predict(self, x):
        x = prepand_one(x)
        return self.inference(x)

def plot_res(plt_x, plt_y):
    fig = plt.figure()
    ax1 = fig.add_subplot(111)
    ax1.scatter(plt_x, plt_y, s=10, c='black', label='real')   
    ax1.scatter(plt_x, model.predict(plt_x), s=10, c='blue', alpha=0.5, label='my')
    ax1.scatter(plt_x, reg.predict(plt_x), s=10, c='red', alpha=0.5, label='sklearn')
